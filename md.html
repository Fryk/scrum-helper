Angular 16 Microfrontend Setup with Module Federation & Shared Component Library

1. Install Required Dependencies

First, install the necessary packages for Module Federation and component library support:

bash
# Module Federation
npm install @angular-architects/module-federation @angular-architects/module-federation-tools --save-dev

# For Bootstrap-based component library with version isolation
npm install @ng-bootstrap/ng-bootstrap bootstrap
npm install css-loader postcss-loader postcss-prefixwrap --save-dev

# Your shared component library (example)
npm install @your-org/shared-components-v1 
# or @your-org/shared-components-v2 depending on microfrontend requirements

2. Configure Module Federation

Generate Module Federation Configuration

bash
ng add @angular-architects/module-federation --project your-app-name --type remote --port 4201

Update webpack.config.js

javascript
const ModuleFederationPlugin = require("@module-federation/webpack");
const postcssLoader = require('postcss-loader');

module.exports = {
  output: {
    uniqueName: "your-microfrontend-name",
    publicPath: "auto",
    scriptType: "text/javascript"
  },
  optimization: {
    runtimeChunk: false,
    splitChunks: false
  },
  module: {
    rules: [
      // CSS preprocessing rule for component library isolation
      {
        test: /\.css$/,
        use: [
          'style-loader',
          'css-loader',
          {
            loader: 'postcss-loader',
            options: {
              postcssOptions: {
                plugins: [
                  // Add prefix to all CSS selectors to prevent conflicts
                  require('postcss-prefixwrap')('.mf-your-app-name'),
                  require('autoprefixer')
                ]
              }
            }
          }
        ]
      },
      // SCSS preprocessing for Bootstrap and component library
      {
        test: /\.scss$/,
        use: [
          'style-loader',
          'css-loader',
          {
            loader: 'postcss-loader',
            options: {
              postcssOptions: {
                plugins: [
                  require('postcss-prefixwrap')('.mf-your-app-name'),
                  require('autoprefixer')
                ]
              }
            }
          },
          'sass-loader'
        ]
      }
    ]
  },
  plugins: [
    new ModuleFederationPlugin({
      name: "your-microfrontend-name",
      filename: "remoteEntry.js",
      exposes: {
        './Component': './src/app/app.module.ts',
        // Add other components/modules you want to expose
      },
      shared: {
        "@angular/core": { 
          singleton: true, 
          strictVersion: true, 
          requiredVersion: "auto" 
        },
        "@angular/common": { 
          singleton: true, 
          strictVersion: true, 
          requiredVersion: "auto" 
        },
        "@angular/common/http": { 
          singleton: true, 
          strictVersion: true, 
          requiredVersion: "auto" 
        },
        "@angular/router": { 
          singleton: true, 
          strictVersion: true, 
          requiredVersion: "auto" 
        },
        // DON'T share component libraries to allow different versions
        // Each microfrontend will bundle its own version
        "bootstrap": { 
          singleton: false,  // Allow multiple versions
          requiredVersion: false 
        }
      },
    }),
  ],
};

4. Component Library Version Management

Create Version-Specific Wrapper Module

Create src/app/shared-components.module.ts:

typescript
import { NgModule } from '@angular/core';
import { CommonModule } from '@angular/common';

// Import your specific version of the component library
import { SharedComponentsModule as SharedComponentsV1 } from '@your-org/shared-components-v1';
// or import { SharedComponentsModule as SharedComponentsV2 } from '@your-org/shared-components-v2';

// Import Bootstrap with version-specific prefix
import { NgbModule } from '@ng-bootstrap/ng-bootstrap';

@NgModule({
  imports: [
    CommonModule,
    SharedComponentsV1, // Use the version your microfrontend needs
    NgbModule
  ],
  exports: [
    SharedComponentsV1,
    NgbModule
  ]
})
export class MicrofrontendSharedModule { }

5. Style Encapsulation (Preventing Style Leaks)

Global Style Configuration

Update src/styles.css:

css
/* Import the isolated Bootstrap styles */
@import './styles/bootstrap-isolated.scss';

/* Root wrapper for the microfrontend */
.mf-your-app-name {
  /* Isolate this microfrontend from global styles */
  all: initial;
  font-family: -apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, sans-serif;
  
  /* Reset common properties that might leak */
  * {
    box-sizing: border-box;
  }
  
  /* Your microfrontend-specific global styles */
  --mf-primary-color: #007bff;
  --mf-secondary-color: #6c757d;
  --mf-border-radius: 4px;
  
  /* Ensure z-index isolation */
  position: relative;
  z-index: 0;
}

/* Prevent Bootstrap classes from affecting other microfrontends */
.mf-your-app-name .btn {
  /* Bootstrap button styles are scoped here */
}

.mf-your-app-name .card {
  /* Bootstrap card styles are scoped here */
}

PostCSS Configuration

Create postcss.config.js in the project root:

javascript
module.exports = {
  plugins: [
    require('postcss-prefixwrap')('.mf-your-app-name'),
    require('autoprefixer')
  ]
};

Component-Level Encapsulation

Ensure all components use ViewEncapsulation:

typescript
import { Component, ViewEncapsulation } from '@angular/core';

@Component({
  selector: 'app-your-component',
  templateUrl: './your-component.component.html',
  styleUrls: ['./your-component.component.css'],
  encapsulation: ViewEncapsulation.Emulated // This is default, ensures style isolation
})
export class YourComponent {
  // Component logic
}

Root Component Wrapper

Update your app.component.html to wrap everything in a unique class:

html
<div class="mf-your-app-name">
  <!-- Your app content -->
  <router-outlet></router-outlet>
</div>

Update app.component.ts

typescript
import { Component, ViewEncapsulation, OnInit } from '@angular/core';

@Component({
  selector: 'app-root',
  templateUrl: './app.component.html',
  styleUrls: ['./app.component.scss'],
  encapsulation: ViewEncapsulation.Emulated
})
export class AppComponent implements OnInit {
  
  ngOnInit() {
    // Apply CSS containment for better isolation
    const rootElement = document.querySelector('.mf-your-app-name');
    if (rootElement) {
      (rootElement as HTMLElement).style.contain = 'layout style paint';
    }
  }
}

5. Bootstrap Configuration

Update main.ts for Native Integration

typescript
import { platformBrowserDynamic } from '@angular/platform-browser-dynamic';
import { AppModule } from './app/app.module';

// Function to mount the microfrontend
export const mount = (element: HTMLElement) => {
  platformBrowserDynamic()
    .bootstrapModule(AppModule)
    .then(moduleRef => {
      console.log('Microfrontend mounted successfully');
    })
    .catch(err => console.error(err));
};

// Auto-mount if running standalone
if (!window.hasOwnProperty('mountMicrofrontend')) {
  mount(document.getElementById('root') || document.body);
}

// Expose mount function globally
(window as any).mountYourAppName = mount;

Update app.module.ts

typescript
import { NgModule, Injector, DoBootstrap } from '@angular/core';
import { BrowserModule } from '@angular/platform-browser';
import { createCustomElement } from '@angular/elements';

import { AppComponent } from './app.component';
import { AppRoutingModule } from './app-routing.module';
import { MicrofrontendSharedModule } from './shared-components.module';

@NgModule({
  declarations: [
    AppComponent
  ],
  imports: [
    BrowserModule,
    AppRoutingModule,
    MicrofrontendSharedModule // Import version-specific component library
  ],
  providers: [],
  entryComponents: [AppComponent]
})
export class AppModule implements DoBootstrap {
  constructor(private injector: Injector) {}

  ngDoBootstrap() {
    // Create custom element for native integration
    const customElement = createCustomElement(AppComponent, {
      injector: this.injector
    });
    customElements.define('your-microfrontend-app', customElement);
  }
}

7. Multi-Version Component Library Strategy

Component Library Wrapper Service

Create src/app/services/component-adapter.service.ts:

typescript
import { Injectable } from '@angular/core';

// Define interface for component library compatibility
export interface ComponentLibraryAdapter {
  getButtonComponent(): any;
  getCardComponent(): any;
  getModalComponent(): any;
  // Add other components as needed
}

@Injectable({
  providedIn: 'root'
})
export class ComponentAdapterService implements ComponentLibraryAdapter {
  
  private version = 'v1'; // or 'v2' based on environment/config
  
  getButtonComponent() {
    // Return version-specific button component
    if (this.version === 'v1') {
      // Return v1 button implementation
      return import('@your-org/shared-components-v1').then(m => m.ButtonComponent);
    } else {
      // Return v2 button implementation
      return import('@your-org/shared-components-v2').then(m => m.ButtonComponentV2);
    }
  }
  
  getCardComponent() {
    if (this.version === 'v1') {
      return import('@your-org/shared-components-v1').then(m => m.CardComponent);
    } else {
      return import('@your-org/shared-components-v2').then(m => m.CardComponentV2);
    }
  }
  
  getModalComponent() {
    if (this.version === 'v1') {
      return import('@your-org/shared-components-v1').then(m => m.ModalComponent);
    } else {
      return import('@your-org/shared-components-v2').then(m => m.ModalComponentV2);
    }
  }
}

javascript
// Dynamic loading
const loadMicrofrontend = async (containerElement) => {
  try {
    // Load the remote entry
    await import('http://localhost:4201/remoteEntry.js');
    
    // Mount the microfrontend
    if (window.mountYourAppName) {
      window.mountYourAppName(containerElement);

  }
};

8. CSS Isolation Best Practices

Component-Level Isolation for Shared Components

typescript
import { Component, ViewEncapsulation, Input } from '@angular/core';

@Component({
  selector: 'app-shared-wrapper',
  template: `
    <div class="mf-component-wrapper">
      <!-- Shared component usage -->
      <shared-button [variant]="buttonVariant" [size]="buttonSize">
        <ng-content></ng-content>
      </shared-button>
    </div>
  `,
  styles: [`
    :host {
      display: block;
      isolation: isolate;
      contain: layout style paint;
    }
    
    .mf-component-wrapper {
      /* Override shared component styles safely */
      --shared-primary-color: var(--mf-primary-color);
      --shared-border-radius: var(--mf-border-radius);
      
      /* Ensure Bootstrap classes are properly scoped */
      .btn {
        /* This .btn will only affect buttons within this component */
      }
      
      .card {
        /* This .card will only affect cards within this component */
      }
    }
  `],
  encapsulation: ViewEncapsulation.Emulated
})
export class SharedComponentWrapper {
  @Input() buttonVariant: string = 'primary';
  @Input() buttonSize: string = 'medium';
}

Dynamic Style Injection for Version Compatibility

Create src/app/services/style-manager.service.ts:

typescript
import { Injectable, Inject, DOCUMENT } from '@angular/core';
import { componentLibraryConfig } from '../environments/component-library.config';

@Injectable({
  providedIn: 'root'
})
export class StyleManagerService {
  
  constructor(@Inject(DOCUMENT) private document: Document) {}
  
  injectVersionSpecificStyles() {
    const version = componentLibraryConfig.version;
    const prefix = componentLibraryConfig.prefix;
    
    // Inject version-specific CSS variables
    const styleElement = this.document.createElement('style');
    styleElement.textContent = `
      .${prefix} {
        /* Version-specific CSS variables */
        --component-lib-version: '${version}';
        --bootstrap-btn-padding-y: ${version === 'v1' ? '0.375rem' : '0.5rem'};
        --bootstrap-btn-padding-x: ${version === 'v1' ? '0.75rem' : '1rem'};
        --bootstrap-border-radius: ${componentLibraryConfig.customTheme.borderRadius};
        
        /* Override component library defaults */
        .shared-button {
          padding: var(--bootstrap-btn-padding-y) var(--bootstrap-btn-padding-x);
          border-radius: var(--bootstrap-border-radius);
        }
        
        /* Version-specific overrides */
        ${version === 'v2' ? `
          .shared-card {
            box-shadow: 0 4px 6px rgba(0, 0, 0, 0.1);
          }
        ` : ''}
      }
    `;
    
    this.document.head.appendChild(styleElement);
  }
  
  removeGlobalBootstrapConflicts() {
    // Remove any global Bootstrap classes that might conflict
    const globalStyles = this.document.querySelectorAll('style, link[rel="stylesheet"]');
    globalStyles.forEach((style) => {
      if (style.textContent?.includes('.btn') || 
          (style as HTMLLinkElement).href?.includes('bootstrap')) {
        // Check if it's not our scoped version
        if (!style.textContent?.includes(`.${componentLibraryConfig.prefix}`)) {
          console.warn('Global Bootstrap detected, may cause conflicts');
        }
      }
    });
  }
}

10. Deployment Considerations

  1. CORS: Ensure proper CORS headers for cross-origin loading
  2. Content Security Policy: Configure CSP to allow microfrontend loading
  3. Versioning: Implement proper versioning strategy for updates
  4. Fallbacks: Implement error boundaries and fallback mechanisms

This setup will create a self-contained Angular microfrontend that builds into minimal files, prevents style leaks, and can be integrated into any host application using native JavaScript.
